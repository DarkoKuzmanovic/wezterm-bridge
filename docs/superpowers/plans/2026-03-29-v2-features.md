# wezterm-bridge v2 Features Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 6 features (wait, read --new, spawn, log, --json, lock/unlock) to transform wezterm-bridge from a text relay into a multi-agent orchestration tool.

**Architecture:** All features are added to the existing single bash script `bin/wezterm-bridge`. New state files (cursors, locks, log) go in the existing `/tmp/wezterm-bridge-$UID/` directory. Each feature gets its own test file. The `--json` flag and `log_event()` are cross-cutting concerns wired into all commands.

**Tech Stack:** Bash (strict mode), WezTerm CLI, jq, coreutils (date, grep, md5sum, tail)

```bash
cmd_wait() {
    require_args 1 $# "wait"
    require_wezterm
    local target
    target="$(resolve_target "$1")"
    shift

    local match="" quiet="" prompt=false
    local timeout=120 interval=2
    local conditions=0

    while [[ $# -gt 0 ]]; do
        case "$1" in
            --match)   match="$2"; (( conditions += 1 )); shift 2 ;;
            --quiet)   quiet="$2"; (( conditions += 1 )); shift 2 ;;
            --prompt)  prompt=true; (( conditions += 1 )); shift ;;
            --timeout) timeout="$2"; shift 2 ;;
            --interval) interval="$2"; shift 2 ;;
            *)         die "unknown flag '$1' for wait" ;;
        esac
    done

    if (( conditions == 0 )); then
        die "wait requires one of: --match <regex>, --quiet <secs>, --prompt"
    fi
    if (( conditions > 1 )); then
        die "wait accepts only one condition (got $conditions)"
    fi

    # Validate numeric args
    if [[ ! "$timeout" =~ ^[0-9]+$ ]]; then
        die "timeout must be a positive integer, got '$timeout'"
    fi
    if [[ ! "$interval" =~ ^[0-9]+$ ]] || [[ "$interval" -eq 0 ]]; then
        die "interval must be a positive integer, got '$interval'"
    fi

    if $prompt; then
        match='^[›$%#>] ?$'
    fi

    if [[ -n "$quiet" ]]; then
        if [[ ! "$quiet" =~ ^[0-9]+$ ]] || [[ "$quiet" -eq 0 ]]; then
            die "quiet seconds must be a positive integer, got '$quiet'"
        fi
    fi

    local start_time
    start_time="$(date +%s)"
    local last_hash="" last_change_time="$start_time"

    while true; do
        local now
        now="$(date +%s)"
        local elapsed=$(( now - start_time ))

        if (( elapsed >= timeout )); then
            log_event ">" "$target" "wait" "timeout=${timeout}s"
            die "wait timed out after ${timeout}s"
        fi

        local content
        content="$(wezterm cli get-text --pane-id "$target" --start-line -50)"

        if [[ -n "$match" ]]; then
            local matched_line
            matched_line="$(printf '%s\n' "$content" | grep -E "$match" | tail -1)" || true
            if [[ -n "$matched_line" ]]; then
                mark_read "$target"
                log_event ">" "$target" "wait" "--match matched=${elapsed}s"
                if [[ "$JSON_OUTPUT" == "true" ]]; then
                    local encoded
                    encoded="$(printf '%s' "$matched_line" | jq -Rs '.')"
                    printf '{"status":"ok","command":"wait","pane_id":"%s","matched_line":%s,"elapsed_seconds":%d}\n' \
                        "$target" "$encoded" "$elapsed"
                else
                    printf '%s\n' "$matched_line"
                fi
                return 0
            fi
        fi

        if [[ -n "$quiet" ]]; then
            local current_hash
            current_hash="$(printf '%s' "$content" | cksum | cut -d' ' -f1)"
            if [[ "$current_hash" != "$last_hash" ]]; then
                last_hash="$current_hash"
                last_change_time="$now"
            fi
            local quiet_elapsed=$(( now - last_change_time ))
            if (( quiet_elapsed >= quiet )); then
                mark_read "$target"
                log_event ">" "$target" "wait" "--quiet ${quiet}s settled=${elapsed}s"
                if [[ "$JSON_OUTPUT" == "true" ]]; then
                    json_ok "wait" "pane_id=$target" "quiet_seconds=$quiet" "elapsed_seconds=$elapsed"
                else
                    echo "pane $target quiet for ${quiet}s (waited ${elapsed}s)"
                fi
                return 0
            fi
        fi

        sleep "$interval"
    done
}
```

read_since_cursor() {
    local pane_id="$1" content="$2"
    local cursor_file="$CURSOR_DIR/$pane_id"
    if [[ ! -f "$cursor_file" ]]; then
        # No cursor — return all content
        printf '%s\n' "$content"
        return
    fi
    local saved_count saved_hash
    IFS=: read -r saved_count saved_hash < "$cursor_file"

    # Split content into lines
    local total_lines
    total_lines="$(printf '%s\n' "$content" | wc -l | tr -d ' ')"

    if (( saved_count > total_lines )); then
        # Content scrolled past — return everything
        printf '%s\n' "$content"
        return
    fi

    # Check if the line at saved_count matches the hash
    local line_at_cursor
    line_at_cursor="$(printf '%s\n' "$content" | sed -n "${saved_count}p")"
    local current_hash
    current_hash="$(printf '%s' "$line_at_cursor" | cksum | cut -d' ' -f1)"

    if [[ "$current_hash" == "$saved_hash" ]]; then
        # Cursor matches — return lines after saved_count
        if (( saved_count < total_lines )); then
            printf '%s\n' "$content" | tail -n +"$(( saved_count + 1 ))"
        else
            echo "(no new output)"
        fi
    else
        # Cursor mismatch — return everything with warning
        echo "[cursor mismatch — showing all content]"
        printf '%s\n' "$content"
    fi
}

    require_args 1 $# "spawn"
    require_wezterm
    local label="$1"; shift
    local cmd="" cwd="" direction="--bottom"

    local new_pane_id
    new_pane_id="$("${args[@]}")"

    # Label the new pane (dedup existing labels)
    ensure_label_dir
    local f
    for f in "$LABEL_DIR"/*; do
        [[ -f "$f" ]] || continue
        if [[ "$(cat "$f")" == "$label" ]]; then
            rm -f "$f"
        fi
    done
    printf '%s\n' "$label" > "$LABEL_DIR/$new_pane_id"

    log_event ">" "-" "spawn" "$label pane=$new_pane_id"
    if [[ "$JSON_OUTPUT" == "true" ]]; then

cmd_lock() {
    require_args 1 $# "lock"
    require_wezterm
    local target
    target="$(resolve_target "$1")"
    local timeout_secs=30

    if [[ "${2:-}" == "--timeout" ]]; then
        timeout_secs="${3:-30}"
        if [[ ! "$timeout_secs" =~ ^[0-9]+$ ]] || [[ "$timeout_secs" -eq 0 ]]; then
            die "timeout must be a positive integer, got '$timeout_secs'"
        fi
    fi

    mkdir -p "$LOCK_DIR"

    # Check existing lock
    if [[ -f "$LOCK_DIR/$target" ]]; then
        local owner ts existing_timeout
        IFS=: read -r owner ts existing_timeout < "$LOCK_DIR/$target"
        local now
        now="$(date +%s)"
        local expires=$(( ts + existing_timeout ))
        if (( now < expires )) && [[ "$owner" != "${WEZTERM_PANE:-}" ]]; then
            local remaining=$(( expires - now ))
            die "pane $target already locked by pane $owner (expires in ${remaining}s)"
        fi
    fi

    # Acquire lock
    local now
    now="$(date +%s)"
    printf '%s:%s:%s\n' "${WEZTERM_PANE:-?}" "$now" "$timeout_secs" > "$LOCK_DIR/$target"

    # Read pane to satisfy guard
    wezterm cli get-text --pane-id "$target" --start-line -50 >/dev/null
    mark_read "$target"

    log_event ">" "$target" "lock" "timeout=${timeout_secs}s"
    if [[ "$JSON_OUTPUT" == "true" ]]; then
        json_ok "lock" "pane_id=$target" "timeout=$timeout_secs"
    else
        echo "locked pane $target (expires in ${timeout_secs}s)"
    fi
}

cmd_unlock() {
    require_args 1 $# "unlock"
    local target
    target="$(resolve_target "$1")"

    mkdir -p "$LOCK_DIR"
    local lock_file="$LOCK_DIR/$target"

    if [[ ! -f "$lock_file" ]]; then
        if [[ "$JSON_OUTPUT" == "true" ]]; then
            json_ok "unlock" "pane_id=$target" "was_locked=false"
        else
            echo "pane $target not locked"
        fi
        return
    fi

    local owner ts timeout_secs
    IFS=: read -r owner ts timeout_secs < "$lock_file"

    if [[ "$owner" != "${WEZTERM_PANE:-}" ]]; then
        local now
        now="$(date +%s)"
        local expires=$(( ts + timeout_secs ))
        if (( now < expires )); then
            die "pane $target locked by pane $owner, cannot unlock"
        fi
    fi

    rm -f "$lock_file"
    log_event ">" "$target" "unlock" ""
    if [[ "$JSON_OUTPUT" == "true" ]]; then
        json_ok "unlock" "pane_id=$target" "was_locked=true"
    else
        echo "unlocked pane $target"
    fi
}

---

## wait <target> <condition>

Blocks until the target pane's content matches a condition. Polls every `--interval` seconds.

Conditions (exactly one required):
- `--match <regex>` — wait for a line matching the extended regex
- `--quiet <secs>` — wait for output to stop changing for N seconds
- `--prompt` — shortcut for `--match '^[›$%#>] ?$'`

Options:
- `--timeout <secs>` — max wait time (default: 120)
- `--interval <secs>` — poll interval (default: 2)

On success: prints the matched line (for --match/--prompt) or a status message, sets read guard, exits 0.
On timeout: exits 1.

```
wezterm-bridge wait codex --match "PASS" --timeout 60
wezterm-bridge wait codex --quiet 10
wezterm-bridge wait codex --prompt
```

---

## read <target> --new

Returns only output that appeared since the last `read` of this target.

Uses a cursor system: after each read, a cursor (line count + hash) is saved. On `--new`, only lines after the cursor are returned. If the cursor can't be matched (scrollback wrapped), all content is returned with a warning.

```
wezterm-bridge read codex --new
```

---

## spawn <label> [options]

Splits the current pane, starts a command, and labels the new pane.

Options:
- `--cmd <command>` — command to run (default: `$SHELL`)
- `--cwd <dir>` — working directory
- `--horizontal` — side-by-side split (default: top/bottom)

```
wezterm-bridge spawn codex --cmd codex --cwd ~/project --horizontal
wezterm-bridge spawn worker --cmd 'npm run dev'
```

---

## log [--tail <n>] [--clear]

View or clear the audit log. All bridge operations are logged automatically.

```
wezterm-bridge log                   # last 20 entries
wezterm-bridge log --tail 50         # last 50 entries
wezterm-bridge log --clear           # clear log
```

---

## lock <target> [--timeout <secs>]

Acquires an advisory lock on a pane. Write commands (`type`, `keys`, `message`) check locks before executing. Locks auto-expire after the timeout.

Default timeout: 30 seconds.

```
wezterm-bridge lock codex --timeout 60
```

---

## unlock <target>

Releases a lock you own on a pane. Idempotent if pane is not locked.

```
wezterm-bridge unlock codex
```

---

## --json (global flag)

Prefix any command with `--json` to get machine-readable JSON output.

```
wezterm-bridge --json list
wezterm-bridge --json read codex 10
wezterm-bridge --json wait codex --match "done"
```

Output format:
```json
{"status":"ok","command":"...","key":"value"}
{"status":"error","message":"..."}
```
```
